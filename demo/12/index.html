class MyComponent extends migi.Component {
  constructor(...data) {
    super(...data);
    this.va = 1;
    this.vb = 2;
  }
  get a() {
    return this.va;
  }
  set a(v) {
    this.va = v;
  }
  get b() {
    return this.vb;
  }
  set b(v) {
    this.vb = v;
  }
  click1() {
    this.a++;
  }
  click2() {
    this.b++;
  }
  render() {
  return (
    <div>
      <p onClick={this.click1}>a++</p>
      <p onClick={this.click2}>b++</p>
      <p>a: {this.a}, b: {this.b}, max: {Math.max(this.a, this.b)}</p>
    </div>
  );
  }
}

migi.render(
  <MyComponent />,
  '#example'
);
