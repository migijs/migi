class MyComponent extends migi.Component {
  constructor(...data) {
    super(...data);
  }
  handleClick() {
    this.element.querySelector('[ref]').focus();
  }
  render() {
    return (
      <div>
        <input type="text" ref="myTextInput" />
        <input type="button" value="Focus the text input" onClick={this.handleClick} />
      </div>
    );
  }
}

migi.render(
  <MyComponent />,
  '#example'
);
