class Input extends migi.Component {
  constructor(...data) {
    super(...data);
    this.text = 'Hello!';
  }
  get txt() {
    return this.text;
  }
  set txt(v) {
    this.text = v;
  }
  render() {
    return (
      <div>
        <input value={this.txt}/>
        <p>{this.txt}</p>
      </div>
    );
  }
}

migi.render(
  <Input />,
  '#example'
);
